cmake_minimum_required(VERSION 3.16.0)

# FOUNDATIONAL FLOOR (jae/floor-handwire, decision #724): the minimal
# bootable base firmware -- bb_log + while(1) print, HAND-WIRED (a la
# carte, no bb_init walker). Every layered-up bb example should be
# measurably larger than this floor.
#
# B1-973 slice 1a: dropped the hand-maintained `set(COMPONENTS ...)`
# allowlist for the generated `BB_AUTOWIRE_COMPONENTS` fragment (B1-747:
# scripts/bbtool/composition.py) that `bbtool codegen` already emits
# alongside `BB_AUTOWIRE_REQUIRES` (see main/CMakeLists.txt) -- one
# manifest edit in bbtool.toml's [capability.floor] now keeps BOTH the
# link-time REQUIRES and the discovery-time COMPONENTS allowlist in
# lockstep, instead of a second hand list to update on every change.
# Without an explicit COMPONENTS allowlist ESP-IDF's component manager
# would build EVERY component found under EXTRA_COMPONENT_DIRS, not just
# floor's real closure -- same reason examples/smoke restricts discovery
# per board, just generated here instead of hand-maintained.
#
# Usage (never `pio ci`):
#   pio run -d . -e esp32
#   python3 ../../scripts/bbtool.py size --build-dir .pio/build/esp32
#
# The codegen path (`bbtool codegen`) mirroring this same component set is
# proven separately via `make floor-codegen` (see repo Makefile) -- it does
# not touch discovery here, and floor's primary build stays hand-wired.

include("${CMAKE_CURRENT_SOURCE_DIR}/../../cmake/bb_component_dirs.cmake")
bb_component_dirs("${CMAKE_CURRENT_SOURCE_DIR}/../../components" BB_COMPONENT_DIRS)
list(APPEND EXTRA_COMPONENT_DIRS
    ${BB_COMPONENT_DIRS}
)

# Same P4-only-target workaround as examples/smoke/CMakeLists.txt: the
# component-manager's manifest-solve pass walks every idf_component.yml under
# EXTRA_COMPONENT_DIRS before COMPONENTS is applied, so bb_display_ek79007's
# esp32p4-only pin must be excluded even though this example never selects it.
set(EXCLUDE_COMPONENTS bb_display_ek79007)

# BB_AUTOWIRE_COMPONENTS (generated by `bbtool codegen` from
# bbtool.toml's [capability.floor] manifest) IS the discovery allowlist --
# must be set before including project.cmake. A missing fragment is a hard
# error (run `make floor-gen` first), matching main/CMakeLists.txt's own
# BB_AUTOWIRE_REQUIRES guard for the same generated file.
include("${CMAKE_CURRENT_SOURCE_DIR}/../../cmake/bb_generated.cmake")
bb_include_generated_or_fail(BB_FLOOR_GENERATED_COMPONENTS
    "${CMAKE_CURRENT_SOURCE_DIR}/main/generated/bb_autowire_components.cmake"
    "run `make floor-gen` (see the Makefile) before building.")
# `__pio_env` is PlatformIO's own auto-generated dummy component, required
# regardless of the manifest -- BB_AUTOWIRE_COMPONENTS (a portable, PIO-
# agnostic codegen artifact) never includes it, so it's added here.
set(COMPONENTS __pio_env ${BB_AUTOWIRE_COMPONENTS})

include($ENV{IDF_PATH}/tools/cmake/project.cmake)
project(floor)

# B1-1045 PR-4: bb_mdns (a new floor addition, the Model-B egress vehicle
# expansion) includes bb_core's bb_hw.h, which hard-requires
# BB_HW_BOARD_HEADER -- floor's single board target is the classic
# ESP32-D0/WROOM-32 (board = esp32dev in platformio.ini). Mirrors
# examples/smoke/CMakeLists.txt's board-header wiring.
idf_build_set_property(COMPILE_OPTIONS "-I${CMAKE_CURRENT_SOURCE_DIR}/board" APPEND)
idf_build_set_property(COMPILE_OPTIONS "-DBB_HW_BOARD_HEADER=\"esp32_wroom_32.h\"" APPEND)
